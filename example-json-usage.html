JavaScript Code to Load JSON:

// Fetch and load products.json
fetch('products.json')
    .then(response => response.json())
    .then(data => {
        console.log('Store:', data.meta.store_name);
        console.log('Total Games:', data.games.length);
        console.log('Total Consoles:', data.consoles.length);
        console.log('Total Accessories:', data.accessories.length);
        
        // Display products
        displayProducts(data);
    })
    .catch(error => console.error('Error loading products:', error));
